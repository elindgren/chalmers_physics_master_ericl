\tikzset{%

  dropout neuron/.style={
    circle,

  },
  input neuron/.style={
    circle,
    draw,
    fill=input_node,
    thick,
    minimum size=1cm
  },
  output neuron/.style={
    circle,
    draw,
    fill=output_node,
    thick,
    minimum size=1cm
  },
  neuron missing/.style={
    draw=none, 
    scale=4,
    fill=none,
    text height=0.333cm,
    execute at begin node=\color{black}$\vdots$
  },
}

\foreach \m [count=\y] in {1,2,3,missing,4}
  \node [dense neuron/.try, neuron \m/.try ] (hidden2-\m) at (8,2.5-\y) {};
